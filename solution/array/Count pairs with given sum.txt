						Count pairs with given sum
--------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
Example 1:

Input:
N = 4, K = 6
arr[] = {1, 5, 7, 1}
Output: 2
Explanation: 
arr[0] + arr[1] = 1 + 5 = 6 
and arr[1] + arr[3] = 5 + 1 = 6.

Example 2:

Input:
N = 4, X = 2
arr[] = {1, 1, 1, 1}
Output: 6
Explanation: 
Each 1 will produce sum 2 with any 1

#include<bits/stdc++.h>
using namespace std;
int getPairs(int arr[],int len,int sum){
    unordered_map<int,int> m;
    int range=0;
    for(int i=0;i<len;i++){
        m[arr[i]]++;
    }
    for(int i=0;i<len;i++){
        range+=m[sum-arr[i]];
        if(sum-arr[i]==arr[i]){
            range--;
        }
    }
    return range/2;

}

int main(){

    int arr[]={1, 5, 7, 1};
    int len =sizeof(arr)/sizeof(arr[0]);
    int sum=6;
    cout<<getPairs(arr,len,sum); 
   
 }
