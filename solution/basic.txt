#include<bits/stdc++.h>
using namespace std;

int main(){
    int n,num,min,max,div,re,temp,flag=0;
    cin>>n;

    while (n--){
        // cin>>num;
        cin>>min;
        cin>>max;
        
        for(int i=min;i<=max;i++){
            int sum=0;
            temp=i;
            while(temp>0){
                re=temp%10;
                sum=sum*10+re;
                temp/=10;
            }
            if(sum==i)   cout<<i<<"-->palindrome"<<"\n";
            // else    cout<<temp<<"-->NO";
        }
    }

    return 0;
    
}

#include<bits/stdc++.h>
using namespace std;

int main(){
    int n,num,min,max,div,re,temp,num2,num1,num3;
    cin>>n;

    while (n--){
        cin>>num;
        num1=0;
        num2=1;
        cout<<0<<"\n"<<1<<"\n";
        for(int i=2;i<num;i++){
            num3=num1+num2;
            cout<<num3<<"\n";
            num1=num2;
            num2=num3;
        }
    }
    
    return 0;
    
}

#include<bits/stdc++.h>
using namespace std;

int main(){
    int n,num,min,max,div,re,fact=1;
    cin>>n;

    while (n--){
        cin>>num;
        for(int i=1;i<=num;i++){
            fact*=i;
        }
        cout<<fact<<" ";
    }
    
    return 0;
    
}
