							Palindrome String

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
#include<bits/stdc++.h>
using namespace std;

int main(){
    string str = "abbaa";
    int flag=0;
    int len=str.size();
    int j=len-1;
    for(int i=0;i<=(str.size()/2)-1;i++){
        if(str[i]!=str[j]){
            flag=1;
            break;
        }
        j--;
    }
    if (flag==0)    cout<<"yes";
    else    cout<<"NO";

}
