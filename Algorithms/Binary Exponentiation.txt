#include<bits/stdc++.h>
using namespace std;
int main(){    
    long long x=3;
    long long y=21;
    long long temp=1;                        
                            
    while(y>0){
        if(y%2==1){
            temp*=x;
            cout<<"temp-->"<<temp<<"\n";
        }       
        x*=x;
        cout<<"x-->"<<x<<"\n";
        y/=2;
    }
    cout<<temp;
}
